# oracle 常用命令记录
查询当前用户信息{
	查看当前用户拥有的角色权限信息=select * from role_sys_privs;
	查看当前用户的详细信息=select * from user_users;
	查看当前用户的角色信息=select * from user_role_privs;
}
执行计划开关{
	SET AUTOTRACE OFF=此为默认值，即关闭Autotrace 
	SET AUTOTRACE ON EXPLAIN=只显示执行计划
	SET AUTOTRACE ON STATISTICS=只显示执行的统计信息
	SET AUTOTRACE ON=包含2,3两项内容
	SET AUTOTRACE TRACEONLY=与ON相似，但不显示语句的执行结果
}
SCN{
	系统当前SCN=select current_scn from v$database;
	系统检查点SCN=select CHECKPOINT_CHANGE# from v$database;
	文件SCN=select file#,checkpoint_change# from v$datafile;
	结束SCN=select TABLESPACE_NAME,STATUS from dba_tablespaces;
	数据文件头SCN=select file#,CHECKPOINT_CHANGE# from v$datafile_header;
}
